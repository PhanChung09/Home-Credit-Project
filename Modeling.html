Modeling Plan and Speed Strategy

This notebook is structured to satisfy the full modeling requirements with a practical runtime strategy:

  • Use 3-fold CV during exploration.
  • Use randomized search (~20 iterations) for tuning.
  • Use a 5K sample for tuning.
  • Train the final model on the full training data.

We evaluate models on AUC (not only accuracy), because target default is highly imbalanced.

prep_glm_df <- function(df, vars) {
  out <- df %>% select(all_of(c("TARGET", vars)))
  for (nm in names(out)) {
    if (is.numeric(out[[nm]])) {
      med <- median(out[[nm]], na.rm = TRUE)
      out[[nm]][is.na(out[[nm]])] <- med
    }
  }
  out
}

# Core logistic
logit_core_df <- prep_glm_df(train_small, core_vars)
mod_core <- glm(as.formula(paste("TARGET ~", paste(core_vars, collapse = " + "))), data = logit_core_df, family = binomial())
prob_core <- predict(mod_core, newdata = logit_core_df, type = "response")
pred_core <- ifelse(prob_core > 0.5, 1, 0)
acc_core <- mean(pred_core == logit_core_df$TARGET)
auc_core <- safe_auc(logit_core_df$TARGET, prob_core)

# Larger logistic

Code 
cv_auc_glm <- function(X_df, y_vec, folds, vars) {
  aucs <- c()
  X_use <- X_df %>% select(all_of(intersect(vars, names(X_df))))

  for (f in seq_along(folds)) {
    te <- folds[[f]]
    tr <- setdiff(seq_len(nrow(X_use)), te)

    tr_df <- X_use[tr, ]
    te_df <- X_use[te, ]
    y_tr <- y_vec[tr]
    y_te <- y_vec[te]

    # median imputation for glm
    for (nm in names(tr_df)) {
      med <- median(tr_df[[nm]], na.rm = TRUE)
      tr_df[[nm]][is.na(tr_df[[nm]])] <- med
      te_df[[nm]][is.na(te_df[[nm]])] <- med
    }

    fit <- glm(TARGET ~ ., data = cbind(TARGET = y_tr, tr_df), family = binomial())
    p <- predict(fit, newdata = te_df, type = "response")
    aucs <- c(aucs, safe_auc(y_te, p))
  }
  aucs
}

cv_auc_rf <- function(X_df, y_vec, folds, ntree = 200) {
  aucs <- c()
  for (f in seq_along(folds)) {
    te <- folds[[f]]
    tr <- setdiff(seq_len(nrow(X_df)), te)

    rf <- randomForest(x = X_df[tr, ], y = as.factor(y_vec[tr]), ntree = ntree)
    p <- predict(rf, X_df[te, ], type = "prob")[, "1"]
    aucs <- c(aucs, safe_auc(y_vec[te], p))
  }
  aucs
}

cv_auc_xgb <- function(X_df, y_vec, folds, params, nrounds = 120) {
  aucs <- c()
  for (f in seq_along(folds)) {
    te <- folds[[f]]
    tr <- setdiff(seq_len(nrow(X_df)), te)

    dtr <- xgb.DMatrix(as.matrix(X_df[tr, ]), label = y_vec[tr])
    dte <- xgb.DMatrix(as.matrix(X_df[te, ]))

    fit <- xgb.train(params = params, data = dtr, nrounds = nrounds, verbose = 0)
    p <- predict(fit, dte)
    aucs <- c(aucs, safe_auc(y_vec[te], p))
  }
  aucs
}

5) Class Imbalance Strategies

We compare no adjustment vs class-weighting vs upsampling vs downsampling.

Code 
# Fixed holdout for quick strategy comparison
idx_imb <- sample(seq_len(nrow(X)), min(10000, nrow(X)))
X_imb <- X[idx_imb, ]
y_imb <- y[idx_imb]

tr_idx <- createDataPartition(y_imb, p = 0.8, list = FALSE)
X_tr <- X_imb[tr_idx, ]
y_tr <- y_imb[tr_idx]
X_va <- X_imb[-tr_idx, ]
y_va <- y_imb[-tr_idx]

# A) No adjustment
fit_a <- xgb.train(
  params = list(objective = "binary:logistic", eval_metric = "auc"),
  data = xgb.DMatrix(as.matrix(X_tr), label = y_tr),
  nrounds = 120,
  verbose = 0
)
auc_a <- safe_auc(y_va, predict(fit_a, xgb.DMatrix(as.matrix(X_va))))

# B) Class weight
spw <- sum(y_tr == 0) / sum(y_tr == 1)
fit_b <- xgb.train(
  params = list(objective = "binary:logistic", eval_metric = "auc", scale_pos_weight = spw),
  data = xgb.DMatrix(as.matrix(X_tr), label = y_tr),
  nrounds = 120,
  verbose = 0
)
auc_b <- safe_auc(y_va, predict(fit_b, xgb.DMatrix(as.matrix(X_va))))

# C) Upsampling
df_tr <- cbind(TARGET = y_tr, X_tr)
up <- ovun.sample(TARGET ~ ., data = as.data.frame(df_tr), method = "over", N = 2 * sum(y_tr == 0))$data
fit_c <- xgb.train(
  params = list(objective = "binary:logistic", eval_metric = "auc"),
  data = xgb.DMatrix(as.matrix(up %>% select(-TARGET)), label = up$TARGET),
  nrounds = 120,
  verbose = 0
)
auc_c <- safe_auc(y_va, predict(fit_c, xgb.DMatrix(as.matrix(X_va))))

# D) Downsampling
down <- ovun.sample(TARGET ~ ., data = as.data.frame(df_tr), method = "under", N = 2 * sum(y_tr == 1))$data
fit_d <- xgb.train(
  params = list(objective = "binary:logistic", eval_metric = "auc"),
  data = xgb.DMatrix(as.matrix(down %>% select(-TARGET)), label = down$TARGET),
  nrounds = 120,
  verbose = 0
)
auc_d <- safe_auc(y_va, predict(fit_d, xgb.DMatrix(as.matrix(X_va))))

imbalance_tbl <- tibble(
  Strategy = c("None", "Class Weight (scale_pos_weight)", "Upsampling", "Downsampling"),
  AUC = c(auc_a, auc_b, auc_c, auc_d)
) %>% arrange(desc(AUC))

6) Hyperparameter Tuning (Randomized Search)

Use 5K rows + 3-fold CV + 20 random iterations.

Code 
idx_tune <- sample(seq_len(nrow(X)), min(5000, nrow(X)))
X_t <- X[idx_tune, ]
y_t <- y[idx_tune]
folds_t <- createFolds(y_t, k = 3, list = TRUE, returnTrain = FALSE)

random_params <- function(y_vec) {
  list(
    objective = "binary:logistic",
    eval_metric = "auc",
    max_depth = sample(3:8, 1),
    eta = runif(1, 0.03, 0.30),
    subsample = runif(1, 0.60, 1.00),
    colsample_bytree = runif(1, 0.60, 1.00),
    min_child_weight = sample(1:10, 1),
    gamma = runif(1, 0, 5),
    scale_pos_weight = sum(y_vec == 0) / sum(y_vec == 1)
  )
}

n_iter <- 20
tune_res <- vector("list", n_iter)

for (i in seq_len(n_iter)) {
  p <- random_params(y_t)
  aucs <- cv_auc_xgb(X_t, y_t, folds_t, p, nrounds = 120)
  tune_res[[i]] <- list(iter = i, params = p, mean_auc = mean(aucs), sd_auc = sd(aucs))
}

tune_tbl <- tibble(
  iter = sapply(tune_res, `[[`, "iter"),
  mean_auc = sapply(tune_res, `[[`, "mean_auc"),
  sd_auc = sapply(tune_res, `[[`, "sd_auc")
) %>% arrange(desc(mean_auc))

best_iter <- tune_tbl$iter[1]
best_params <- tune_res[[best_iter]]$params

tune_tbl %>% slice_head(n = 10)
