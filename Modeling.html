Modeling Overview

This notebook builds a practical modeling workflow for Home Credit Default Risk with a strong focus on out-of-sample AUC, because the target is highly imbalanced (~8% default rate).

The notebook covers:

  1. Baseline benchmark (majority class and simple logistic regression)
  2. Candidate model comparison with 3-fold cross-validation
  3. Class imbalance strategies (none vs. SMOTE vs. downsampling)
  4. Randomized hyperparameter search for the best model
  5. Impact of supplementary features
  6. Kaggle submission generation

To keep runtime manageable, hyperparameter tuning uses a 5K sample, 3-fold CV, and randomized search with ~20 iterations.

To keep runtime fast, we use a small predictor subset on a 5K sample.

# Sample for quick baseline
sample_idx <- sample(1:nrow(train), min(5000, nrow(train)))
train_small <- train[sample_idx, ]

# Select a minimal predictor set if available
candidate_vars <- c(
  "EXT_SOURCE_1", "EXT_SOURCE_2", "EXT_SOURCE_3",
  "AMT_CREDIT", "AMT_INCOME_TOTAL", "AMT_ANNUITY",
  "DAYS_BIRTH", "DAYS_EMPLOYED"
)

use_vars <- intersect(candidate_vars, names(train_small))

logit_df <- train_small %>%
  select(all_of(c("TARGET", use_vars)))

# Basic NA handling for logistic model
for (col in names(logit_df)) {
  if (is.numeric(logit_df[[col]])) {
    logit_df[[col]][is.na(logit_df[[col]])] <- median(logit_df[[col]], na.rm = TRUE)
  }
}

# Fit logistic model
formula_logit <- as.formula(paste("TARGET ~", paste(use_vars, collapse = " + ")))
logit_model <- glm(formula_logit, data = logit_df, family = binomial())

logit_probs <- predict(logit_model, newdata = logit_df, type = "response")
logit_preds <- ifelse(logit_probs > 0.5, 1, 0)

logit_accuracy <- mean(logit_preds == logit_df$TARGET)
logit_auc <- as.numeric(auc(logit_df$TARGET, logit_probs))

cat("Logistic baseline accuracy:", round(logit_accuracy, 4), "\n")

Code 
cv_auc_logistic <- function(X_df, y_vec, folds) {
  aucs <- c()
  for (i in seq_along(folds)) {
    test_idx <- folds[[i]]
    train_idx <- setdiff(seq_len(nrow(X_df)), test_idx)

    X_train <- X_df[train_idx, ]
    y_train <- y_vec[train_idx]
    X_test <- X_df[test_idx, ]
    y_test <- y_vec[test_idx]

    # Median imputation for logistic
    for (col in names(X_train)) {
      med <- median(X_train[[col]], na.rm = TRUE)
      X_train[[col]][is.na(X_train[[col]])] <- med
      X_test[[col]][is.na(X_test[[col]])] <- med
    }

    train_df <- cbind(TARGET = y_train, X_train)
    model <- glm(TARGET ~ ., data = train_df, family = binomial())
    probs <- predict(model, newdata = X_test, type = "response")

    aucs <- c(aucs, as.numeric(auc(y_test, probs)))
  }
  aucs
}

cv_auc_rf <- function(X_df, y_vec, folds, ntree = 200) {
  aucs <- c()
  for (i in seq_along(folds)) {
    test_idx <- folds[[i]]
    train_idx <- setdiff(seq_len(nrow(X_df)), test_idx)

    X_train <- X_df[train_idx, ]
    y_train <- as.factor(y_vec[train_idx])
    X_test <- X_df[test_idx, ]
    y_test <- y_vec[test_idx]

    rf_model <- randomForest(
      x = X_train,
      y = y_train,
      ntree = ntree,
      importance = FALSE
    )

    probs <- predict(rf_model, X_test, type = "prob")[, "1"]
    aucs <- c(aucs, as.numeric(auc(y_test, probs)))
  }
  aucs
}

cv_auc_xgb <- function(X_df, y_vec, folds, params = list()) {
  aucs <- c()

  # Default fast params
  default_params <- list(
    objective = "binary:logistic",
    eval_metric = "auc",
    max_depth = 6,
    eta = 0.1,
    subsample = 0.8,
    colsample_bytree = 0.8
  )
  params <- modifyList(default_params, params)

  for (i in seq_along(folds)) {
    test_idx <- folds[[i]]
    train_idx <- setdiff(seq_len(nrow(X_df)), test_idx)

    X_train <- as.matrix(X_df[train_idx, ])
    y_train <- y_vec[train_idx]
    X_test <- as.matrix(X_df[test_idx, ])
    y_test <- y_vec[test_idx]

    dtrain <- xgb.DMatrix(data = X_train, label = y_train)
    dtest <- xgb.DMatrix(data = X_test)

    xgb_model <- xgb.train(
      params = params,
      data = dtrain,
      nrounds = 120,
      verbose = 0
    )

    probs <- predict(xgb_model, dtest)
    aucs <- c(aucs, as.numeric(auc(y_test, probs)))
  }
  aucs
}
